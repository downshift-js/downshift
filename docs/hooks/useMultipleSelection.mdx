import {useState} from 'react'
import {Playground} from 'docz'
import {useCombobox, useMultipleSelection} from '../../src'
import {items, menuStyles, comboboxStyles, playgroundStyles} from '../utils'

# useCombobox

## Introduction

<Playground style={playgroundStyles}>
  {() => {
    const DropdownMultipleCombobox = () => {
      const [inputItems, setInputItems] = useState(items)
      const {getSelectedItemProps, getDropdownProps, addItem, items: selectedItems} = useMultipleSelection()
      const {
        isOpen,
        selectedItem,
        getToggleButtonProps,
        getLabelProps,
        getMenuProps,
        getInputProps,
        getComboboxProps,
        highlightedIndex,
        getItemProps,
        setInputValue,
      } = useCombobox({
        items: inputItems,
        onInputValueChange: ({inputValue}) => {
          setInputItems(
            items.filter(item =>
              item.toLowerCase().startsWith(inputValue.toLowerCase()),
            ),
          )
        },
        onSelectedItemChange: ({selectedItem}) => {
          setInputValue('')
          addItem(selectedItem)
        },
      })

      return (
        <>
          <label {...getLabelProps()}>Choose an element:</label>
          <div style={{display: 'flex', flexWrap: 'wrap'}}>
            {selectedItems.map((selectedItem, index) => (
              <span {...getSelectedItemProps({item: selectedItem, index})}>{selectedItem}</span>
            ))}
            <div style={comboboxStyles} {...getComboboxProps()}>
              <input {...getInputProps({...getDropdownProps()})} />
            </div>
          </div>
          <button {...getToggleButtonProps()} aria-label={'toggle menu'}>
            &#8595;
          </button>
          <ul {...getMenuProps()} style={menuStyles}>
            {isOpen &&
              inputItems.map((item, index) => (
                <li
                  style={
                    highlightedIndex === index
                      ? {backgroundColor: '#bde4ff'}
                      : {}
                  }
                  key={`${item}${index}`}
                  {...getItemProps({item, index})}
                >
                  {item}
                </li>
              ))}
          </ul>
        </>
      )
    }

    return <DropdownMultipleCombobox />

}}

</Playground>
